Assignment

                
$a = 1;
$b = 'Hello';
$c = ['array','of','items'];
$d = ['key' => 'value']; // Uses arrays as dictionaries
$e = true; // 'True' also accepted
                
            
                
a = 1
b = 'Hello'
c = ['array','of','items']
d = {'key': 'value'}
e = True # 'true' is an error
                
            
                
let a = 1; // Or 'const a = 1;' to prevent re-assignment
let b = 'Hello'; // $b (etc) is allowed but uncommon
let c = ['array','of','items'];
let d = {key: 'value'};
let e = true; // 'True' is an error

Arithmetic

                
$a = 1;
$a++; // 2
$a+=2; // 4
$a--; // 3

$a = 4 / 2; // 2
$a = 3 % 3; // 1 (modulo)
$a = 3 ** 2; // 9 (power)
                
            
                
a = 1
a += 1 # 2 - no '++' operator
a += 2 # 4
a -= 1 # 3

a = 4 / 2 # 2.0
a = 3 % 2 # 1 (modulo)
a = 3 ** 2 # 9 (power)
                
            
                
let a = 1;
a++; //2
a+=2; //4
a--; //3

a = 4 / 2 // 2.0
a = 3 % 2 // 1 (modulo)
a = 3 ** 2 // 9 (power)

Strings

message = f'{greeting} {subject}";

let greeting = 'Hello';
let subject = 'World';

Functions

                
function increase($number, $by=1) {
  return $number + $by;
}

increase(7); // 8
increase(27, 15); // 42

Conditions

                
if ($a === 42) {
  print('You have the answer');
} else if ($a > 42) {
  print('Too high!');
} else {
  print('Too low!');
}
                
            
                
if a == 42:
  print('You have the answer');
elif a > 42:
  print('Too high!')
else:
  print('Too low!')
                
            
                
if (a === 42) {
  console.log('You have the answer');
} else if (a > 42) {
  console.log('Too high!');
} else {
  console.log('Too low!');
}

Loop in range

                
for ($i=10; $i>0; $i--) {
  print("$i green bottles, standing on a wall\n");
}
                
            
                
for i in range(10,0,-1):
  print(f"{i} green bottles, standing on a wall")
                
            
                
for (i=10; i>0; i--) {
  console.log(i + " green bottles, standing on a wall");
}
